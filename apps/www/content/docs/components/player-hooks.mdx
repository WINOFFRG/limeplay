<Steps>
<Step>

```npm
npx shadcn add @limeplay/player-hooks
```
</Step>

<Step>
### Setup the player hooks in provider

```tsx lineNumbers title="components/limeplay/media-player.tsx"
import { PlayerHooks } from "@/components/limeplay/player-hooks" // [!code ++]

// ... other imports

export function MediaPlayer() {
  return (
    <MediaProvider>
      // [!code ++]
      <PlayerHooks />
      <Suspense>
        <Layout.RootContainer height={720} width={1280}>
          <Layout.PlayerContainer>
            <MediaElement />
            <Layout.ControlsContainer>
              <BottomControls />
            </Layout.ControlsContainer>
          </Layout.PlayerContainer>
        </Layout.RootContainer>
      </Suspense>
    </MediaProvider>
  )
}
```
